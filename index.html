#include #include using namespace std; void main() { srand(time(NULL)); const size_t n = 14, m = 3; int arr[n][m]; for (size_t k = 0; k < m; ++k) for (size_t i = 0; i < n; ++i) { arr[k][i] = rand() % 15 + 1; for (size_t j = 0; j < i; ++j) { if (arr[k][j] == arr[k][i]) { --i; break; } } } for (size_t i = 0; i < n; ++i) { cout << i + 1 << "\t"; for (size_t k = 0; k < m; ++k) cout << arr[k][i] << "\t"; cout << endl; } }
